import { Variants } from '@/components/Variants'
import { Disabling } from '@/components/Disabling'

export const classes = {
  plugin: function () {
    return function ({ addUtilities }) {
      addUtilities({
        '.backdrop-filter': { filter: 'var(--tw-backdrop-blur) var(--tw-backdrop-brightness) var(--tw-backdrop-contrast) var(--tw-backdrop-grayscale) var(--tw-backdrop-hue-rotate) var(--tw-backdrop-invert) var(--tw-backdrop-opacity) var(--tw-backdrop-saturate) var(--tw-backdrop-sepia)' },
        '.backdrop-filter-none': { filter: 'none' },
      })
    }
  }
}

## 用法

使用 `backdrop-filter` 來啟用背景濾鏡 ( 跟其他的背景濾鏡功能共用，如 `backdrop-blur` or `backdrop-grayscale`)。而 `backdrop-filter-none` 功能是可移除濾鏡的.

```html
<div class="backdrop-filter backdrop-grayscale backdrop-blur-md backdrop-contrast-200 ...">
  <!-- ... -->
</div>
```

## 響應式

如要將一個元素的背景亮度控制在特定斷點，在任何一個既有背景亮度功能的前綴加上 `{screen}:` prefix to any existing backdrop filter utility. For example, use `md:backdrop-filter-none` to apply the `backdrop-filter-none` utility at only medium screen sizes and above.

```html
<div class="backdrop-filter backdrop-blur-lg **md:backdrop-filter-none** ...">
  <!-- ... -->
</div>
```

想知道更多有關 Tailwind 響應式設計功能的訊息，請參考 [響應式設計] (/docs/responsive-design) 文件
